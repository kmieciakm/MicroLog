query {
    dailyLogs (
        order: { timestamp: DESC, level: DESC }
        where: {
            level: { in: [CRITICAL, ERROR, WARNING] }
        }
    ) {
        items {
            identity {
                eventId
            },
            timestamp,
            message,
            level,
            levelName,
            exception {
                type,
                source,
                message,
                innerException {
                    message
                }
            },
            properties {
                name,
                value
            }
        }
    }
}
